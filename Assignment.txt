CREATE DATABASE ORDERS




 -- 1 --
CREATE TABLE CUSTOMER
(
CUSTOMERID INT IDENTITY PRIMARY KEY ,
FISTNAME VARCHAR (40) NOT NULL,
LASTNAME VARCHAR(40),
CITY VARCHAR(40),
COUNTRY VARCHAR(40),
PHONE VARCHAR(20),
 CUSTOMERNAME AS FISTNAME +' ' + LASTNAME
)


CREATE INDEX CUSTOMERNAME ON CUSTOMER(CUSTOMERNAME)




-------------------------------------------------


CREATE TABLE PRODUCT
(
PRODUCTID INT IDENTITY PRIMARY KEY ,
PRODUCTNAME VARCHAR (50),
UNITPRICE DECIMAL(12,2),
PACKAGE VARCHAR(30),
)


CREATE INDEX PRODUCTNAME ON PRODUCT(PRODUCTNAME)


-------------------------------------------------
CREATE TABLE ORDERS
(
ORDERID INT IDENTITY PRIMARY KEY ,
ORDERDATE DATETIME NOT NULL,
ORDERNO NVARCHAR (10),
CUSTOMERID int foreign key  references CUSTOMER (CUSTOMERID),        
TOTALAMOUNT DECIMAL(12,2),
)
CREATE INDEX ORDERCUSTOMERID ON ORDERS(CUSTOMERID)
CREATE INDEX ORDERORDERDATE ON ORDERS(ORDERDATE)


-------------------------------------------------
CREATE TABLE ORDERITEM
(
ORDERITEMID INT IDENTITY PRIMARY KEY ,
ORDERID int foreign key  references ORDERS (ORDERID),        
PRODUCTID int foreign key  references PRODUCT (PRODUCTID),        
UNITPRICE  DECIMAL(12,2),
QUANTITY INT ,
)


CREATE INDEX ORDERITEMORDERID ON ORDERITEM(ORDERID)
CREATE INDEX ORDERITEMPRODUCTID ON ORDERITEM(PRODUCTID)


-------------------------------------------------
